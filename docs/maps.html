# 1. Summarise top-30 port codes and volumes
top30 <- df_1 %>%
  mutate(port_code = str_extract(`Port of Entry`, "^[0-9]{4}")) %>%
  group_by(port_code) %>%
  summarise(Volume = sum(`Sum of Volume`, na.rm = TRUE), .groups="drop") %>%
  arrange(desc(Volume)) %>%
  slice_head(n = 30)

# 3. Combine volumes & coords
df_plot <- left_join(top30, coords30, by = "port_code")

# 4. Warn if any missing
missing <- filter(df_plot, is.na(lon))
if(nrow(missing)) {
  warning("Missing coords for codes: ", paste(missing$port_code, collapse=", "))
}

# 5. Color palette
pal <- colorNumeric("Reds", domain = df_plot$Volume)

# 6. Build interactive map
leaflet(df_plot) %>%
  addProviderTiles("CartoDB.Positron") %>%
  setView(lng = -95, lat = 55, zoom = 4) %>%
  addCircleMarkers(
    ~lon, ~lat,
    radius      = ~Volume / 800000,
    weight      = 1,
    color       = "white",
    fillColor   = ~pal(Volume),
    fillOpacity = 0.8,
    label       = ~paste0(port_code, ": ", scales::comma(Volume)),
    labelOptions= labelOptions(direction = "auto")
  ) %>%
  addLegend(
    "bottomright",
    pal      = pal,
    values   = ~Volume,
    title    = "Total Crossings",
    labFormat= labelFormat(big.mark = ",")
  )
